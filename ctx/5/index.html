⚡ THIS WEEK'S PATTERN

Route cheap tasks to Flash, not Pro

Gemini 3.5 Flash benchmarks ahead of 3.1 Pro at 4x the output speed and half the price — which means the only reason to reach for Pro is when the task genuinely needs it. Audit your agent pipeline: classification, summarization, and structured extraction almost never do. Reserve the heavy model for reasoning chains, multi-step planning, and anything where a wrong answer is expensive to catch downstream.

const MODEL_ROUTING = {
  classify: 'gemini-3.5-flash',
  summarize: 'gemini-3.5-flash',
  extractJson: 'gemini-3.5-flash',
  planSteps: 'gemini-3.5-pro',
  reasonChain: 'gemini-3.5-pro',
  validateOutput: 'gemini-3.5-pro',
} as const;

type TaskType = keyof typeof MODEL_ROUTING;

function getModel(task: TaskType): string {
  return MODEL_ROUTING[task];
}
