⚡ THIS WEEK'S PATTERN

Route cheap tasks to Flash, save heavy models for generation

Gemini 3.5 Flash is faster than any current frontier model at under half the cost — but only if you route deliberately. High-volume, low-stakes tasks (dedup checks, relevance scoring, category tagging) don't need your strongest model. Reserve it for generation steps where output variance actually matters.

const AGENT_MODELS: Record<string, string> = {
  dedupe: 'gemini-flash-3.5',
  relevance: 'gemini-flash-3.5',
  categorise: 'gemini-flash-3.5',
  summarise: 'gemini-pro-2.5',
  lede: 'gemini-pro-2.5',
  slackCopy: 'gemini-flash-3.5',
};

function getModel(agent: string): string {
  return AGENT_MODELS[agent] ?? 'gemini-pro-2.5';
}
